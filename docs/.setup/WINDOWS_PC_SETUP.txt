# Windows PC — full setup guide for mopt-bot
#
# Covers: cloning the repo, Python environment, backend config,
#         running the API, Cloudflare Tunnel, and optional auto-start.
#
# Assumptions: Windows 10/11, PowerShell, local admin access for some steps.
# Commands are shown for PowerShell unless noted otherwise.

# =============================================================================
# 1. SYSTEM PACKAGES
# =============================================================================

# Install Git + Python 3.10+ + Node.js 20+ if missing.
# Recommended installers:
#   Git:    https://git-scm.com/download/win
#   Python: https://www.python.org/downloads/windows/
#   Node:   https://nodejs.org/
#
# Optional package manager approach:
#   winget install --id Git.Git -e
#   winget install --id Python.Python.3.13 -e
#   winget install --id OpenJS.NodeJS.LTS -e

# Verify:
python --version
git --version
node --version
npm --version


# =============================================================================
# 2. CLONE THE REPOSITORY
# =============================================================================

# Option A — HTTPS
git clone https://github.com/<your-org>/mopt-bot.git
cd mopt-bot

# Option B — SSH
#   ssh-keygen -t ed25519 -C "your@email.com"
#   type $env:USERPROFILE\.ssh\id_ed25519.pub
#   Add key in GitHub, then:
# git clone git@github.com:<your-org>/mopt-bot.git
# cd mopt-bot


# =============================================================================
# 3. PYTHON VIRTUAL ENVIRONMENT
# =============================================================================

# Create repo-local venv (project convention):
python -m venv venv

# Activate (optional; direct venv\Scripts\python.exe also works):
.\venv\Scripts\Activate.ps1

# Install backend + solver dependencies:
.\venv\Scripts\python.exe -m pip install --upgrade pip
.\venv\Scripts\pip.exe install -r requirements.txt


# =============================================================================
# 4. BACKEND CONFIGURATION (.env)
# =============================================================================

# The app checks backend/.env first, then repo-root .env.
Copy-Item backend\.env.example backend\.env
notepad backend\.env

# Important variables:
#
#   MOPT_HOST=127.0.0.1
#   MOPT_PORT=8000
#   MOPT_PUBLIC_URL=https://mopt.your-domain.example
#   MOPT_CLIENT_SECRET=<strong-secret>
#   MOPT_RESEARCHER_SECRET=<strong-secret>
#   MOPT_DEFAULT_GEMINI_MODEL=gemini-3-flash-preview
#
#   MOPT_DATABASE_URL=sqlite:///./data/mopt_study.db
#   MOPT_SOLVE_TIMEOUT_SEC=120
#
#   MOPT_CORS_ORIGINS=https://your-frontend-domain.com
#   (for local frontend dev, include localhost:5173 as needed)
#
# Optional Fernet key:
#   .\venv\Scripts\python.exe -c "from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())"
#   Paste into MOPT_FERNET_KEY=...


# =============================================================================
# 5. RUN THE API
# =============================================================================

cd backend

# Production:
..\venv\Scripts\python.exe .\run_server.py

# Development (auto-reload):
# ..\venv\Scripts\python.exe .\run_server.py --reload

# Health check (new terminal):
curl http://127.0.0.1:8000/health


# =============================================================================
# 6. CLOUDFLARE TUNNEL — INSTALL cloudflared
# =============================================================================

# Option A — winget:
winget install --id Cloudflare.cloudflared -e

# Option B — manual:
# Download from https://developers.cloudflare.com/cloudflare-one/connections/connect-networks/downloads/
# and install cloudflared for Windows.

# Verify:
cloudflared --version


# =============================================================================
# 7. CLOUDFLARE TUNNEL — CREATE VIA DASHBOARD (recommended)
# =============================================================================

# 1) Cloudflare dashboard -> Zero Trust -> Networks -> Tunnels
# 2) Create tunnel -> Cloudflared -> name it (e.g. mopt-backend or mopt2-backend)
# 3) Copy the generated connector command for Windows.
#
# Typical form:
#   cloudflared service install <long-token>
#
# Run elevated PowerShell as Administrator:
cloudflared service install <long-token>

# 4) In dashboard, add Public Hostname:
#    Subdomain: mopt (or mopt2 for backup)
#    Domain:    your-domain.example
#    Service:   http://127.0.0.1:8000
#
# 5) Start/verify service:
sc.exe query cloudflared

# Test externally:
curl https://mopt.your-domain.example/health


# =============================================================================
# 8. OPTIONAL: AUTO-START BACKEND ON WINDOWS
# =============================================================================

# Option A (simple): Task Scheduler on logon/startup
#   Action:
#     Program:  C:\Users\<you>\git-repo\mopt-bot\venv\Scripts\python.exe
#     Args:     run_server.py
#     Start in: C:\Users\<you>\git-repo\mopt-bot\backend
#
# Option B: NSSM (Non-Sucking Service Manager) for true Windows service
#   nssm install mopt-backend
#   (set executable, args, and working directory as above)


# =============================================================================
# 9. FRONTEND DEPLOY/POINTING
# =============================================================================

# If frontend is on Vercel:
# - set VITE_API_BASE to your tunnel URL (e.g. https://mopt.your-domain.example)
# - redeploy frontend
#
# If frontend already has wrong backend saved in browser:
# - use Backend chip and set correct URL
# - or click "Use automatic" to clear local override


# =============================================================================
# 10. FAILOVER / BACKUP TUNNEL NOTES
# =============================================================================

# Recommended setup:
# - mopt  = primary (e.g. Raspberry Pi)
# - mopt2 = backup/testing (Windows PC)
#
# Do not run the same tunnel endpoint from two hosts unless intentional.
# For predictable behavior, run one active connector per hostname target.


# =============================================================================
# 11. TROUBLESHOOTING
# =============================================================================

# - "Frontend blank / cannot login":
#   Usually wrong backend URL or failed health. Use Backend chip on login screen.
#
# - CORS errors:
#   Ensure MOPT_CORS_ORIGINS includes your Vercel domain exactly (no trailing slash).
#
# - Port conflict:
#   Change MOPT_PORT and tunnel service target together.
#
# - Service logs:
#   cloudflared Windows service logs can be inspected in Event Viewer or by
#   running cloudflared in foreground temporarily for diagnostics.
#
# - Updating backend code:
#   cd C:\Users\<you>\git-repo\mopt-bot
#   git pull
#   .\venv\Scripts\pip.exe install -r requirements.txt
#   Restart backend process/service.
